using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading.Tasks;
using Microsoft.AspNetCore.Http;
using Microsoft.AspNetCore.Mvc;
using Microsoft.EntityFrameworkCore;
using Library1.Models;
using WebApplication1.Models;

namespace TestProject
{
    [Route("api/[controller]")]
    [ApiController]
    public class TestApiController : ControllerBase
    {
        private readonly CarContext context;

        public TestApiController(CarContext context)
        {
            this.context = context;
        }

        // GET: api/TestApi
        [HttpGet]
        public async Task<ActionResult<IEnumerable<Car>>> GetCar()
        {
            return await this.context.Car.ToListAsync();
        }

        // GET: api/TestApi/5
        [HttpGet("{id}")]
        public async Task<ActionResult<Car>> GetCar(string id)
        {
            var car = await this.context.Car.FindAsync(id);

            if (car == null)
            {
                return NotFound();
            }

            return car;
        }

        // PUT: api/TestApi/5
        [HttpPut("{id}")]
        public async Task<IActionResult> PutCar(string id, Car car)
        {
            if (id != car.ID)
            {
                return BadRequest();
            }

            this.context.Entry(car).State = EntityState.Modified;

            try
            {
                await this.context.SaveChangesAsync();
            }
            catch (DbUpdateConcurrencyException)
            {
                if (!CarExists(id))
                {
                    return NotFound();
                }
                else
                {
                    throw;
                }
            }

            return NoContent();
        }

        // POST: api/TestApi
        [HttpPost]
        public async Task<ActionResult<Car>> PostCar(Car car)
        {
            this.context.Car.Add(car);
            await this.context.SaveChangesAsync();

            return CreatedAtAction("GetCar", new { id = car.ID }, car);
        }

        // DELETE: api/TestApi/5
        [HttpDelete("{id}")]
        public async Task<ActionResult<Car>> DeleteCar(string id)
        {
            var car = await this.context.Car.FindAsync(id);
            if (car == null)
            {
                return NotFound();
            }

            this.context.Car.Remove(car);
            await this.context.SaveChangesAsync();

            return car;
        }

        private bool CarExists(string id)
        {
            return this.context.Car.Any(e => e.ID == id);
        }
    }
}
